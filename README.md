# QUICK CALCULATION

This library will help to calculate numbers quickly.

# How to install?
'''
pip install quickcal
'''

# Initializing package
'''
from quickcal import QuickCal

num_1 = 16
num_2 = 8

cal = QuickCal(num_1, num_2)
'''

# For Addition
'''
cal.addition()
'''

# For Substraction
'''
cal.substraction()
'''

# For Multiplication
'''
cal.multiplication()
'''

# For Division
'''
cal.division()
'''
